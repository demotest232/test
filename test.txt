import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
import java.lang.reflect.Method;

class ParametersMessageGeneratorTest {

    @Test
    void testCreateMessage_NullArray() throws Exception {
        ParametersMessageGenerator generator = new ParametersMessageGenerator();
        Method method = ParametersMessageGenerator.class.getDeclaredMethod("createMessage", Object[].class);
        method.setAccessible(true);

        CharSequence result = (CharSequence) method.invoke(generator, (Object) null);
        assertEquals("", result.toString());
    }

    @Test
    void testCreateMessage_EmptyArray() throws Exception {
        ParametersMessageGenerator generator = new ParametersMessageGenerator();
        Method method = ParametersMessageGenerator.class.getDeclaredMethod("createMessage", Object[].class);
        method.setAccessible(true);

        CharSequence result = (CharSequence) method.invoke(generator, (Object) new Object[0]);
        assertEquals("", result.toString());
    }

    @Test
    void testCreateMessage_NonEmptyArray() throws Exception {
        ParametersMessageGenerator generator = new ParametersMessageGenerator();
        Method method = ParametersMessageGenerator.class.getDeclaredMethod("createMessage", Object[].class);
        method.setAccessible(true);

       
        Object[] input = new Object[] { "test1", "test2" };
      

        CharSequence result = (CharSequence) method.invoke(generator, (Object) input);
       
       assertTrue(result.toString().contains("test1"));
    }
}
