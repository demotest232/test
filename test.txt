

    @Test
    void testCreateMessage_variousInputs() {
      
        ParametersMessageGenerator generator = new ParametersMessageGenerator();

     
        Object[] params = {
            42,
            "hello",
            Arrays.asList(1, 2, 3),
            new int[]{4, 5, 6},
            null
        };

        CharSequence result = generator.createMessage(params);
        String strResult = result.toString();

       

      
        assertTrue(strResult.contains("42"));
        assertTrue(strResult.contains("hello"));
        assertTrue(strResult.contains("Collection"));
        assertTrue(strResult.contains("Array"));
        assertTrue(strResult.contains("null") || strResult.contains("Null"));

      
    }

    @Test
    void testCreateMessage_emptyInput() {
        ParametersMessageGenerator generator = new ParametersMessageGenerator();
        Object[] params = {};
        CharSequence result = generator.createMessage(params);
        assertEquals("", result.toString());
    }

    @Test
    void testCreateMessage_nullInput() {
        ParametersMessageGenerator generator = new ParametersMessageGenerator();
        CharSequence result = generator.createMessage(null);
        assertEquals("", result.toString());
    }
