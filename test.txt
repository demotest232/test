import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class SoapHeaderRequestScopeUnitTest {

    @Test
    void testSetAndGetClientUser() {
      
        SoapHeaderRequestScope scope = new SoapHeaderRequestScope();
        ClientUser user = new ClientUser(); 
      
        scope.setClientUser(user);

       
        assertEquals(user, scope.getClientUser());
    }

    @Test
    void testGetClientUserWhenNotSet() {
        SoapHeaderRequestScope scope = new SoapHeaderRequestScope();
        assertNull(scope.getClientUser());
    }
}
