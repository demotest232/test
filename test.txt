 public void saveAllFiles(String branch, String location) throws IOException {
        List<String> allFilesList = Arrays.asList("Grants", "Packages", "Views",
                "Functions", "Procedures", "Triggers");

        for (String file : allFilesList) {
            try {
                // Get directory/tree recursively from GitHub
                PagedIterable<GHContent> contents = repository.getDirectoryContent(file, branch);

                for (GHContent content : contents) {
                    if (content.isFile()) {
                        // Create local file structure
                        File outputFile = new File(location + File.separator + content.getPath());
                        outputFile.getParentFile().mkdirs();

                        try (OutputStream os = new FileOutputStream(outputFile, false)) {
                            byte[] decoded = content.read().readAllBytes();
                            os.write(decoded);
                        }
                    }
                }
            } catch (GHFileNotFoundException e) {
                System.out.println(file + " Not Found!!");
            }
        }
    }
