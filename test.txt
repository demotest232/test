@Test
void testConfigurationsEnum() {
    for (RequestDetailConfiguration.Configurations configEnum : RequestDetailConfiguration.Configurations.values()) {
        RequestDetailConfiguration conf = configEnum.getConfiguration();
        assertNotNull(conf);
     
        if (configEnum == RequestDetailConfiguration.Configurations.PMC_PORTFOLIO) {
            assertEquals("pmcapi.portfolio_request_detail", conf.requestDetailTabeName);
            assertEquals("pmcapi.request_detail_seq", conf.sequenceName);
        }
      
    }
}
