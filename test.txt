GHCompare compare = repo.getCompare(tags[0], tags[1]);
Set<String> added = new HashSet<>();
Set<String> removed = new HashSet<>();
Set<String> modified = new HashSet<>();

for (GHCompare.CommitFile f : compare.getFiles()) {
    String status = f.getStatus(); // "added", "removed", "modified", "renamed"
    switch (status) {
        case "added":    added.add(f.getFileName()); break;
        case "removed":  removed.add(f.getFileName()); break;
        case "modified": modified.add(f.getFileName()); break;
        case "renamed":
            // you can handle rename specially if needed:
            // f.getFileName() is new name, f.getPreviousFileName() is old name
            break;
    }
}
