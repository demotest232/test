import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;

public class AuthTokenUnavailableUnitTest {

    @Test
    public void testDefaultConstructor() {
        AuthTokenUnavailable ex = new AuthTokenUnavailable();
        assertNull(ex.getMessage());
        assertNull(ex.getCause());
    }

    @Test
    public void testConstructorWithMessage() {
        String message = "Token is unavailable";
        AuthTokenUnavailable ex = new AuthTokenUnavailable(message);
        assertEquals(message, ex.getMessage());
        assertNull(ex.getCause());
    }

    @Test
    public void testConstructorWithCause() {
        Throwable cause = new RuntimeException("underlying");
        AuthTokenUnavailable ex = new AuthTokenUnavailable(cause);
        assertEquals("java.lang.RuntimeException: underlying", ex.getMessage());
        assertEquals(cause, ex.getCause());
    }

    @Test
    public void testConstructorWithMessageAndCause() {
        String message = "Token Error";
        Throwable cause = new RuntimeException("details");
        AuthTokenUnavailable ex = new AuthTokenUnavailable(message, cause);
        assertEquals(message, ex.getMessage());
        assertEquals(cause, ex.getCause());
    }
}
