private String decodeBlob(GHBlob blob) {
    if ("base64".equalsIgnoreCase(blob.getEncoding())) {
        return new String(Base64.getDecoder().decode(blob.getContent()), StandardCharsets.UTF_8);
    }
    return blob.getContent();
}
