public void generateDiffFile(final File diffFile,
                             final String fromTag,
                             final String toTag,
                             final String compPath) throws IOException {

    // Get repo
    GHRepository repo = github.getOrganization(inputData.getOrgName())
                              .getRepository(inputData.getRepoName());

    // Compare tags
    GHCompare compare = repo.getCompare(fromTag, toTag);

    for (GHCompare.CommitFile file : compare.getFiles()) {
        String fileName = file.getFileName()
                              .substring(file.getFileName().lastIndexOf("/") + 1);

        if (fileName.equalsIgnoreCase(diffFile.getName())) {
            try (OutputStream os = new FileOutputStream(diffFile, false)) {
                if (file.getPatch() != null) {
                    os.write(file.getPatch().getBytes(StandardCharsets.UTF_8));
                }
            }
        }
    }
}
