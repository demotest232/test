import org.junit.jupiter.api.Test;
import java.sql.Timestamp;
import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

class CurrentTimestampValueCallbackTest {

    @Test
    void returnsCurrentTimestamp() {
        GridRow row = mock(GridRow.class);
        CurrentTimestampValueCallback callback = new CurrentTimestampValueCallback();
        long before = System.currentTimeMillis();

        Timestamp ts = callback.value(row);

        long after = System.currentTimeMillis();
        assertNotNull(ts);
        // The timestamp should be between before and after times
        assertTrue(ts.getTime() >= before && ts.getTime() <= after,
                   "Timestamp is not within expected range");
    }
}
